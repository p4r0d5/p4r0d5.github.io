# fix the entropy requirement

# check keys
gpg --list-secret-keys --keyid-format LONG youremail@yourdomain

# show key
gpg --armor --export (id)

# add to git
git config --global user.signingkey (sec ID)

# fix passphrase issue
gpg agent never asks for passphrase.
